{# 模板宏定义 #}

模板宏

{# 模板宏定义：宏相当于函数，调用模板宏传入不同参数可以产生不同内容 #} {% macro input(label="", type="text", name="", value="") %} {% endmacro %}
{{ input("用户名：", name="username") }}
{{ input("密码：", type="password", name="password") }}
{{ input("重复密码：", type="password", name="password2") }}
{{ input(type="submit", value="注册") }}

使用其他模板中定义宏

{% import "macro.html" as func %}
{{ func.input("用户名：", name="username") }}
{{ func.input("密码：", type="password", name="password") }}
{{ func.input("重复密码：", type="password", name="password2") }}
{{ func.input(type="submit", value="注册") }}
